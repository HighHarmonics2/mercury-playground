// generate a small theme from a sine function 
// and transform it with various methods like
// invert(), join() and clone()

set tempo 98

ring theme sine(16 5 0 24)
ring variation invert(theme 0 24)
ring phrase join(theme variation)
ring section clone(phrase 0 5 9 7)

new synth triangle name(arpy)
    set arpy time(1/16) shape(1 170) note(section 1)

    // give the synth random panning
    set arpy pan(random)
